request_security_no_repaint(timeframe, expression) =>
    request.security(symbol = syminfo.tickerid, timeframe = timeframe, expression = expression[barstate.isrealtime ? 1 : 0], gaps = barmerge.gaps_off)[barstate.isrealtime ? 0 : 1] // PineCoders best practice.

pad_leading_zeroes(int value, int length) =>
    string result = str.tostring(value)
    while str.length(result) < length
        result := "0" + result
    result

time_add_minutes(int value, int minutes) =>
    int result = value
    int hh = value / 100
    int mm = value - (hh * 100)
    mm += minutes
    if mm > 59
        hh += 1
        mm -= 60
    else if mm < 0
        hh -= 1
        mm += 60
    result := (hh * 100) + mm

build_orb_session(string session, int orb_duration_minutes) =>
    string orb_session = na
    int start_time = int(str.tonumber(str.substring(session, 0, 4)))
    //log.info("START_TIME: [" + str.tostring(start_time) + "]")
    int end_time = time_add_minutes(start_time, orb_duration_minutes)
    //log.info("END_TIME: [" + str.tostring(end_time) + "]")
    orb_session := pad_leading_zeroes(start_time, 4) + "-" + pad_leading_zeroes(end_time, 4)
    //log.info("ORB_SESSION: [" + orb_session + "]")
    orb_session

if show_nyk_orb
    // Check for entering Opening Range
    if na(time("", nyk_orb_session, i_nyk_tz, 1)) and not na(time("", nyk_orb_session, i_nyk_tz))
        // Just entered Opening Range
        nyk_orb_closed := false
        nyk_orb_h := high
        nyk_orb_l := low
        nyk_orb_m := (nyk_orb_h + nyk_orb_l) / 2.0
        if i_show_nyk_orb_box
            nyk_orb_box := box.new(bar_index, nyk_orb_h, bar_index, nyk_orb_l, color.new(i_nyk_orb_color, SESSION_BORDER_TRANSPARENCY), 1, line.style_solid, extend.none, xloc.bar_index, color.new(i_nyk_orb_color, SESSION_BACKGROUND_TRANSPARENCY), str.tostring(i_nyk_orb_duration) + "m ORB", size.normal, color.new(i_nyk_orb_color, SESSION_BORDER_TRANSPARENCY), text.align_center, text.align_top)
    else if not na(time("", nyk_orb_session, i_nyk_tz))
        // In Opening Range
        nyk_orb_h := math.max(high, nyk_orb_h)
        nyk_orb_l := math.min(low, nyk_orb_l)
        nyk_orb_m := (nyk_orb_h + nyk_orb_l) / 2.0
        if i_show_nyk_orb_box
            nyk_orb_box.set_right(bar_index)
            nyk_orb_box.set_top(nyk_orb_h)
            nyk_orb_box.set_bottom(nyk_orb_l)
    else if not na(time("", nyk_orb_session, i_nyk_tz, 1)) and na(time("", nyk_orb_session, i_nyk_tz))
        // Just left Opening Range
        nyk_orb_closed := true
        lbl_nyk_orb_h := (show_nyk_orb and i_show_nyk_orb_h and i_show_labels) ? label.new(bar_index, nyk_orb_h, str.tostring(i_nyk_orb_duration) + "m ORB High", xloc.bar_index, yloc.price, color.new(color.black, 100), label.style_label_left, i_nyk_orb_h_color, size.normal, text.align_left, str.tostring(i_nyk_orb_duration) + "m Opening Range High") : na
